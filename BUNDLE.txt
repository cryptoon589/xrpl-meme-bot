================================================================================
XRPL MEME BOT v1.6.0 - Complete Source Bundle
================================================================================

To use:
1. Create directory: mkdir xrpl-meme-bot && cd xrpl-meme-bot
2. Copy each file section below into its respective path
3. Run: npm install && cp .env.example .env && npm run build && npm start

FILE: .env.example
================================================================================
# XRPL Meme Bot Configuration

# XRPL Connection
XRPL_WS_URL=wss://rpc.xrplclaw.com/ws

# Telegram Bot
TELEGRAM_BOT_TOKEN=your_bot_token_here
TELEGRAM_CHAT_ID=your_chat_id_here

# Operating Mode: WATCH or PAPER (AUTO is stub-only)
MODE=WATCH

# Risk Thresholds
MIN_LIQUIDITY_XRP=2000
MIN_SCORE_ALERT=75
MIN_SCORE_PAPER_TRADE=80

# Paper Trading Config
STARTING_BANKROLL_XRP=500
MAX_TRADE_XRP=20
MAX_OPEN_TRADES=3
MAX_DAILY_LOSS_XRP=50

# Scoring Weights (should sum to 100)
WEIGHT_LIQUIDITY=20
WEIGHT_HOLDER_GROWTH=15
WEIGHT_BUY_PRESSURE=20
WEIGHT_VOLUME_ACCEL=15
WEIGHT_DEV_SAFETY=15
WEIGHT_WHITELIST_BOOST=10
WEIGHT_SPREAD=5

# Logging
LOG_LEVEL=info
